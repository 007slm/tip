Tip - Atip 演示（默认样式和结构，更漂亮的 API）

10点钟位置

seajs.use(['atip'], function(Atip) {
    var t = new Atip({
        trigger: '#test',
        content: '<div style="padding:10px">我是内容</div>',
        arrowPosition: 10
    });
});

7点钟位置，更改内容

seajs.use(['atip'], function(Atip) {
    var t = new Atip({
        trigger: '#test6',
        content: '<div style="padding:10px">我是内容</div>',
        theme: 'blue',
        arrowPosition: 7
    });
    t.set('content', '更改后的内容');
});

本来是7点钟的Tip，但位置太靠上了就会变成11点的Tip

seajs.use(['atip'], function(Atip) {
    var t = new Atip({
        trigger: '#test8',
        height: 100,
        width: 200,
        content: '<div>比较高的内容</div>',
        theme: 'white',
        inViewport: true,
        arrowPosition: 7
    });
});

本来是1点钟的Tip，但位置太靠下了就会变成5点的Tip

点击弹出Tip

seajs.use(['atip'], function(Atip) {
    var t = new Atip({
        trigger: '#test7',
        triggerType: 'click',        
        content: '<div style="padding:10px">我是内容</div>',
        theme: 'white',
        arrowPosition: 7
    });
});
